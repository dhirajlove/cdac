Q.4
#include<stdio.h>
int main()
{
	int arr[10]={1 ,2 ,3 ,4 ,5 };
	int *ptr=&arr[8];
	printf("-%d", *(ptr-5));
   	return 0;
}
A. 4
B. -3
C. -5
D. -4
Answer: D

// &arr[0] = 400 (assume)
// &arr[8] = 432 (8 ints will take 32 bytes, so base address + 32)
// ptr = &arr[8] = 432
// ptr - 5 = ptr - 5 * scale factor(ptr) = ptr - 5 * 4 = 432 - 20 = 412
// *(ptr - 5) = * 412 = 4
// -4

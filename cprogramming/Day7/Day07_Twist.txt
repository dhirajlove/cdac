			
#include <stdio.h>		
int main( void )
{
		int num1,num2,num3, num4;

		num1 = 100 ; num2 = 200 ;

		num2= ++num1;
		// num2 = 101 
		// num1 = 101 

		num3= num2--/2;
		num3= 101 / 2 --> 50 --> num3 = 50 
		num2--> num2-- -> 100 

		// num2 = 100 num3=50 num1=101 

		// -%10d --> - - - - - - - - - - + 2 spaces --> 12  		

		num4 = printf(" %-10d ", ++num1 ) + ++num2; // num1 --> 102 
		//num4 = 12 + ++num2-> 12 + 101 --> 113
		//      (10+2spaces)  		
		//++num1 --> 102 
		// ++num2 --> 101   


		printf("%10d\t",num3);// 50

		printf("%-10d",num4); // 113 
 
		return 0;
}
